1. SSH Configuration

First, open the ssh config file:

notepad .ssh\config 
or
code .ssh\config

Add the following to the config file:

Host ibex
  HostName glogin.ibex.kaust.edu.sa
  User USERNAME

Important: Replace "USERNAME" with your actual IBEX username.

2. Generate SSH Key

Generate an SSH key on your laptop:

ssh-keygen -t rsa -b 4096

Copy the contents of the generated public key (usually id_rsa.pub).

3. Connect to IBEX

Connect to IBEX using the following command:

ssh ibex

4. Add SSH Key to Authorized Keys

After connecting to IBEX, add your key to the authorized_keys file:

cd ~/.ssh
nano authorized_keys

Paste your public key into this file and save.

5. Install Miniconda

Download and install Miniconda:

wget https://repo.anaconda.com/miniconda/Miniconda3-latest-Linux-x86_64.sh
bash Miniconda3-latest-Linux-x86_64.sh -b
~/miniconda3/bin/conda init bash && source ~/.bashrc && conda config --set auto_activate_base false

6. Create GPU Job Script

Create a script for running GPU jobs:

nano srun_gpu.sh

Paste the following content into the file:

#!/bin/bash

# Color definitions
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[0;33m'
BLUE='\033[0;34m'
PURPLE='\033[0;35m'
CYAN='\033[0;36m'
NC='\033[0m' # No Color

# Default values
gpu_type=${1:-rtx2080ti}
time=${2:-03:00:00}
mem=${3:-20G}
num_gpus=${4:-1}

# Function to print help message
print_help() {
    echo -e "${BLUE}Usage:${NC} gpu [gpu_type] [time] [memory] [num_gpus]"
    echo -e "       gpu [command]"
    echo
    echo -e "${GREEN}Parameters:${NC}"
    echo -e "  ${YELLOW}gpu_type${NC}  : Type of GPU ${CYAN}(default: rtx2080ti)${NC}"
    echo -e "  ${YELLOW}time${NC}      : Job duration ${CYAN}(default: 03:00:00)${NC}"
    echo -e "  ${YELLOW}memory${NC}    : Memory allocation ${CYAN}(default: 20G)${NC}"
    echo -e "  ${YELLOW}num_gpus${NC}  : Number of GPUs ${CYAN}(default: 1)${NC}"
    echo
    echo -e "${GREEN}Commands:${NC}"
    echo -e "  ${YELLOW}list${NC}      : List your running jobs"
    echo -e "  ${YELLOW}cancel${NC}    : Cancel a specific job"
    echo -e "  ${YELLOW}cancel-all${NC}: Cancel all your jobs"
    echo
    echo -e "${PURPLE}Examples:${NC}"
    echo -e "  gpu                     ${CYAN}# Use all default values${NC}"
    echo -e "  gpu v100 04:00:00 32G 2 ${CYAN}# 2 V100 GPUs, 4 hours, 32GB memory${NC}"
    echo -e "  gpu - - - 4             ${CYAN}# 4 GPUs with default type, time, and memory${NC}"
    echo -e "  gpu list                ${CYAN}# List your running jobs${NC}"
    echo -e "  gpu cancel 12345        ${CYAN}# Cancel job with ID 12345${NC}"
    echo
    echo -e "${RED}Note:${NC} Use '-' to skip a parameter and use its default value."
}

# Function to list jobs
list_jobs() {
    echo -e "${BLUE}Your running jobs:${NC}"
    squeue -u $USER
}

# Function to cancel a job
cancel_job() {
    if [ -z "$1" ]; then
        echo -e "${RED}Error: Please provide a job ID to cancel${NC}"
        return 1
    fi
    echo -e "${YELLOW}Cancelling job $1...${NC}"
    scancel "$1"
    echo -e "${GREEN}Job $1 cancelled${NC}"
}

# Function to cancel all jobs
cancel_all_jobs() {
    echo -e "${YELLOW}Cancelling all your jobs...${NC}"
    scancel -u $USER
    echo -e "${GREEN}All jobs cancelled${NC}"
}

# Check for commands
case "$1" in
    list)
        list_jobs
        exit 0
        ;;
    cancel)
        cancel_job "$2"
        exit 0
        ;;
    cancel-all)
        cancel_all_jobs
        exit 0
        ;;
    -h|--help)
        print_help
        exit 0
        ;;
esac

# Replace '-' with default values
[[ "$gpu_type" == "-" ]] && gpu_type="rtx2080ti"
[[ "$time" == "-" ]] && time="03:00:00"
[[ "$mem" == "-" ]] && mem="20G"
[[ "$num_gpus" == "-" ]] && num_gpus="1"

# Run the srun command
srun --time=$time --gres=gpu:$gpu_type:$num_gpus --mem=$mem --resv-ports=1 --pty /bin/bash -ls

Make the script executable:

chmod +x srun_gpu.sh

7. Create Jupyter Notebook Script

Create a script for starting Jupyter Notebook:

nano start_jupyter.sh

Paste the following content into the file:

#!/bin/bash

# Get the current IP address
IP=$(hostname -I | awk '{print $1}')

# Start Jupyter Notebook with the current IP address and token
jupyter notebook --no-browser --ip=$IP

Make the script executable:

chmod +x start_jupyter.sh

8. Add Aliases to .bashrc

Add aliases for the GPU and Jupyter scripts:

echo "alias gpu='bash ~/srun_gpu.sh'" >> ~/.bashrc
echo "alias start_jupyter='bash ~/start_jupyter.sh'" >> ~/.bashrc
source ~/.bashrc

9. Set Up Conda Environment

Follow these steps to create a new Conda environment with PyTorch and Jupyter:

  1. Create a new Conda environment named 'torch':

    conda create -n torch -y

  2. Activate the new environment:

    conda activate torch

  3. Install Jupyter in the environment:

    conda install jupyter

  4. Install PyTorch with CUDA support:

    conda install pytorch torchvision torchaudio pytorch-cuda=11.8 -c pytorch -c nvidia

    Note: The specific version of PyTorch and CUDA may need to be adjusted based on your system requirements and the latest available versions.

After completing the setup, you can use the following commands to activate or deactivate the environment:

To activate the environment:

conda activate torch

To deactivate the environment when you're done:

conda deactivate

10. Using the Setup

To use a GPU:

gpu

For help with GPU options:

gpu --help

To start Jupyter Notebook:

start_jupyter
